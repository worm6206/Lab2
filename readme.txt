//Compile

./GLCOMPILE.sh lab2.c lab2

//Run

./lab2

//Controls

1,2,3,4 	= 	right arm (clockwise & counterclockwise on upper wing and lower wing)
7,8,9,0 	= 	left arm
w or f 		= 	forward
s or b 		= 	backward
a or l 		= 	left
d or r 		= 	right
q 		= 	exit
c 		= 	reset
t,T		= 	body rotate

//Note

* All rotations are 15 degree (PI/12)
* No collision detection involved
